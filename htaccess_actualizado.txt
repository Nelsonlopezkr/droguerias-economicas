# ══════════════════════════════════════════════════════════
# Droguerías Económicas — .htaccess
# Hostinger / Apache
# Actualizado: Junio 2026
# ══════════════════════════════════════════════════════════

# ── HTTPS forzado ──
RewriteEngine On
RewriteCond %{HTTPS} off
RewriteRule ^(.*)$ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]

# ── WWW → sin www ──
RewriteCond %{HTTP_HOST} ^www\.(.+)$ [NC]
RewriteRule ^(.*)$ https://%1/$1 [R=301,L]

# ── GZIP / compresión ──
<IfModule mod_deflate.c>
  AddOutputFilterByType DEFLATE text/html text/css text/javascript application/javascript application/json
</IfModule>

# ── Cache control ──
<IfModule mod_expires.c>
  ExpiresActive On
  # HTML — cache corto (el catálogo cambia frecuente)
  ExpiresByType text/html                  "access plus 1 hour"
  # CSS y JS — cache largo con versioning
  ExpiresByType text/css                   "access plus 1 month"
  ExpiresByType application/javascript     "access plus 1 month"
  ExpiresByType text/javascript            "access plus 1 month"
  # Imágenes
  ExpiresByType image/webp                 "access plus 6 months"
  ExpiresByType image/jpeg                 "access plus 6 months"
  ExpiresByType image/png                  "access plus 6 months"
  ExpiresByType image/svg+xml              "access plus 1 month"
  # Fuentes
  ExpiresByType font/woff2                 "access plus 1 year"
</IfModule>

# ── Seguridad: bloquear archivos sensibles ──
# CRÍTICO: nunca exponer datos de costos, márgenes ni scripts internos
<FilesMatch "\.(xlsx|xlsm|csv|py|sh|env|log|sql|bak|git)$">
  Order Allow,Deny
  Deny from all
</FilesMatch>

# Bloquear acceso a archivos de configuración
<FilesMatch "^(\.htaccess|\.htpasswd|\.git|web\.config|composer\.json|package\.json)$">
  Order Allow,Deny
  Deny from all
</FilesMatch>

# Bloquear directorio .git completo
RedirectMatch 404 /\.git

# ── Página 404 personalizada ──
ErrorDocument 404 /404.html

# ── Seguridad de headers ──
<IfModule mod_headers.c>
  Header set X-Content-Type-Options "nosniff"
  Header set X-Frame-Options "SAMEORIGIN"
  Header set X-XSS-Protection "1; mode=block"
  Header set Referrer-Policy "strict-origin-when-cross-origin"
  # Cache-busting para JS de catálogo (precios cambian)
  <FilesMatch "productos-data\.js$">
    Header set Cache-Control "no-cache, must-revalidate"
  </FilesMatch>
</IfModule>
